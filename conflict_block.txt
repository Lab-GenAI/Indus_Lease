            print(f"[EXTRACT] Failed to parse batch JSON response: {raw_content[:200]}")
            return [{"name": tag["name"], "value": "Extraction Error"} for tag in tags_batch]

        results = []
        for tag in tags_batch:
            value = parsed.get(tag["name"], "Not Found")
            if not isinstance(value, str):
                value = str(value) if value is not None else "Not Found"
            value = value.strip() if value else "Not Found"
            if _is_garbled_value(value):
                print(f"[EXTRACT] Garbled value detected for '{tag['name']}': {value[:80]} — treating as Not Found")
                value = "Not Found"
            results.append({"name": tag["name"], "value": value})

        return results

    except Exception as e:
        print(f'[EXTRACT] Batch extraction failed: {e}')
        return [{"name": tag["name"], "value": "Extraction Error"} for tag in tags_batch]
